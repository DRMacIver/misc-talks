import pulp

def pack_knapsack(items, capacity):
    problem = pulp.Problem()
    inclusions = [
        pulp.LpVariable('I%d', cat='Binary')
        for i in range(len(items))
    ]
    problem.addConstraint(sum(
      weight * included for weight, _ in items) <= capacity)
    problem.sense = pulp.constants.LpMaximize
    problem.objective = sum(value * included for _, value in items)
    problem.solve()
    return [
      item for item, included in zip(items, inclusions)
      if pulp.value(included) > 0
    ]
